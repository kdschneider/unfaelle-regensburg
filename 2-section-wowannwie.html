## markers ----
getMarkerColor <- function(data) {
  map(
    data$severity,
    function(x) {
      if (x == "Toedlich") {
        "red"
      } else if (x == "Schwer") {
        "orange"
      } else if (x == "Leicht") {
        "beige"
      }
    }
  )
}

icons <- reactive({
  awesomeIcons(
    icon = 'ios-close',
    iconColor = 'black',
    library = 'ion',
    markerColor = getMarkerColor(filtered_data())
  )
})
  
output$map1 <- 
  renderLeaflet({
    leaflet(
      data = filtered_data(),
      options = leafletOptions(
        crs = leafletCRS(code = "WGS84"),
        preferCanvas = NULL
      )
    ) |> 
    fitBounds(~min(lng), ~min(lat), ~max(lng), ~max(lat)) |> 
    setMaxBounds(~min(lng), ~min(lat), ~max(lng), ~max(lat)) |> 
    addProviderTiles(
      provider = providers$OpenStreetMap.DE,
      options = providerTileOptions(minZoom = 11)
    ) |> 
    # leafletProxy funktioniert nicht. die lösung hier ist nicht optimal. 
    # muss immer die gesamte karte neu rendern...
    addAwesomeMarkers(
      group = "Marker",
      lng = ~lng,
      lat = ~lat,
      clusterOptions = markerClusterOptions(),
      icon = icons()
    )
  })

output$monthly_plot <- 
  renderPlot({
    data <- filtered_data()
    
    data |> 
      count(month, .data[[input$fill]]) |> 
      ggplot(
        aes(x = month, y = n)
      ) +
      geom_col(
        aes_string(fill =  input$fill)
      ) +
      theme(
        legend.position = "top",
        legend.title = element_blank(),
        axis.title = element_blank()
      )
  })

plotOutput("monthly_plot")
